C++ — String Manipulation 
// String manipulation examples
#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main() {
    string msg = "hello, world";
    cout << "Length: " << msg.length() << endl;

    transform(msg.begin(), msg.end(), msg.begin(), ::toupper);
    cout << msg << endl; // HELLO, WORLD

    return 0;
}
